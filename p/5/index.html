// fixed header table
.tbOuterDiv {
  position: relative;
  padding-top: 37px;   //height of your thead
}
.tbInnerDiv {
  overflow-y: auto;
  height: 200px;       //the actual scrolling container
}

table.fixedHeader thead {
  display: block;
  position: absolute;
  top: 0;
  left: 0;
}

这样就可以把 thead 固定，而 tabke 翻滚了。下一步，就是要把 th 的宽度用 js 动态处理了！（保持列宽度上下一致）

function scrollingTableSetThWidth(tableId)
{
    var table = document.getElementById(tableId);

    ths = table.getElementsByTagName('th');
    tds = table.getElementsByTagName('td');

    if(ths.length > 0) {
        for(i=0; i < ths.length; i++) {
            tds[i].style.width = getCurrentComputedStyle(ths[i], 'width');
        }
    }
}

function getCurrentComputedStyle(element, attribute)
{
    var attributeValue;
    if (window.getComputedStyle) 
    { // class A browsers
        var styledeclaration = document.defaultView.getComputedStyle(element, null);
        attributeValue = styledeclaration.getPropertyValue(attribute);
    } else if (element.currentStyle) { // IE
        attributeValue = element.currentStyle[vclToCamelCases(attribute)];
    }
    return attributeValue;
}

碳排放与转基因

最后更新: Wed Sep 10 2014 12:37:50 GMT+0800

人类起源

以前 人类起源的说法（国内版本）大致是：多地起源。即，世界各地的猿（人）独自进化行成各自的民族和国家。

九十年代以后，越来越多的证据（化石、基因分析等）表明：人类全部来自非洲某一地区，即智人，。他们迁移到世界各地，取代 而不是杂交了那些本地人科“动物”（如亚洲的直立人、欧洲的尼安德特人等）。

人科分为：南方古猿、能人、直立人和智人。时代不同，物种不同，不能交配，智人与现代人为一个种。

80年代，美国伯克利威尔逊小组发现（线粒体DNA）：全世界有一个（夏娃）共同的女性祖先。

90年代，中国的陈竺和金力指出亚洲人与亚洲直立人没有直接关系，现代中国人来自5万年前的非洲。

转基因

转基因起源于 79年代左右。简单地说，就是人类（而不是上帝或者其它什么神仙）有意识、有方向地创造新的生物或者决定它们的进化方向。好坏目前难以评估，人类已经打开了潘朵拉魔盒。

有三组美国科学家对墨西哥的非转基因玉米（原产地）进行了测试，只有一组没有发现来自孟山都的人为基因片段（也可能是样本数量不足）。也就是说，很快，大自然中将没有原始（生）的玉米种群了，都将会被转基因的玉米杂交。

不只小崔的纪录片，2010我在美国旅游期间，据个人经验，大部门食品的确有绿色（也就是小崔说的有机）标志。而我的美国朋友们，的确不买那些非绿色的便宜食品。

我支持小崔。我不想要任何人想给我的答案，我只想自己去判断。

碳排放

碳排放 是假命题吗？

大部分人，包括罗胖（思维）根本就不清楚，地球温度4－5度的变化意味着什么。金星的命运，包括地球自己的生命起落。

1896 年瑞典科学家就提出了二氧化碳导致温度升高的问题。目前，玉米乙醇计划（经过20年的考证），也在美国被证明是严重的错误（消耗了几乎差不多的石油，包括化肥、运输、加工各环节的石油消耗，而得到了20倍的二氧化碳和盈余一点的能量）。
